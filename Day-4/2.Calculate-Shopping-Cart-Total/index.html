        const cartItems = [
        {
          name: "Laptop",
          price: 999.99
        },
        {
          name: "Headphones",
          price: 149.95
        },
        {
          name: "Mouse",
          price: 29.99
        }
      ];

      let sum = 0;

      for (let i = 0; i < cartItems.length; i++) {
        sum = sum + cartItems[i].price;
      }

      console.log(sum);
