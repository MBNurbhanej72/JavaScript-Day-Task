        const input = document.getElementById("myInput");
        const dataContain = document.getElementById("dataContain");
    
        dataContain.innerText = "No Data Found!";
    
        function addValue() {
    
          if (input.value.trim().length === 0) {
            alert("No Text Added");
            return;
          }
    
          if (dataContain.innerText === "No Data Found!") {
            dataContain.innerText = "";
          }
    
          const createDiv = document.createElement("div");
    
          dataContain.append(createDiv);
    
          var id = Math.random().toString(16).slice(2);
    
          const myHTMLCode = `
          <div id=${id} class="valueDiv">
              <i class="fa-solid fa-trash" onclick="removeData('${id}')" style="cursor:pointer"></i>
              <p id="pTag">${input.value}</p>
          </div>
          `;
    
          createDiv.insertAdjacentHTML("beforeend", myHTMLCode);
    
          input.value = "";
    
        }
    
        function removeData(id) {
          const task = document.getElementById(id);
          if (task) {
            task.remove();
          }
    
          if (dataContain.querySelectorAll(".valueDiv").length === 0) {
            dataContain.innerText = "No Data Found!";
          }
        }
